[2669 : 직사각형 네개의 합집합의 면적 구하기](https://www.acmicpc.net/problem/2669)

## 풀이

으아아아ㅏ

## 코드

```cpp
#include <cstdio>
int k, i, j, x1, x2, y1, y2, a[101][101];
int main() {
    for (; k < 4; k++) {
        scanf("%d %d %d %d", &x1, &y1, &x2, &y2);
        for (i = x1; i < x2; i++) for (j = y1; j < y2; j++) a[i][j] = 1;
    }
    for (i = 1; i <= 100; i++) for (j = 1; j <= 100; j++) k += a[i][j];
    printf("%d", k - 4);
    return 0;
}
```
