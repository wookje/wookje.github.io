[14465 : 소가 길을 건너간 이유 5](https://www.acmicpc.net/problem/14465)  

## 풀이

졸리당

## 코드

```cpp
#include <cstdio>
int n, m, k, cnt, res = 1e9, a[200001];
int main() {
    scanf("%d %d %d", &n, &m, &k);
    while (k--) {
        int x;
        scanf("%d", &x);
        a[x+m] = 1;
    }
    for (int i = 1; i <= n; i++) {
        cnt += a[i+m], cnt -= a[i];
        if (i >= m && cnt < res) res = cnt;
    }
    printf("%d", res);
    return 0;
}
```
