[2740 : 행렬 곱셈](https://www.acmicpc.net/problem/2740)  

## 풀이

으갸갸갸갸갸갸가가가각

## 코드

```cpp
#include <cstdio>
int n, m, k, a[102][102], b[102][102];
int main() {
	scanf("%d%d",&n,&m);
	for (int i = 0; i < n; i++)
        for (int j = 0; j < m; j++)
            scanf("%d", &a[i][j]);
	scanf("%d %d", &m, &k);
    for (int i = 0; i < m; i++)
        for (int j = 0; j < k; j++)
            scanf("%d", &b[i][j]);
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < k; j++) {
			int sum = 0;
			for (int p = 0; p < m; p++) {
				sum += a[i][p]*b[p][j];
			}
			printf("%d ",sum);
		}
		puts("");
	}
    return 0;
}
```
