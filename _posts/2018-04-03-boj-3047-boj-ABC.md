## 코드

```cpp
#include <cstdio>
#include <algorithm>
int a[5];
char b[5];
int main() {
    scanf("%d %d %d %s", &a[0], &a[1], &a[2], b);
    std::sort(a, a + 3);
    printf("%d %d %d", a[b[0] - 'A'], a[b[1] - 'A'], a[b[2] - 'A']);
    return 0;
}
```
