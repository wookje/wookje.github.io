## 풀이

.

## 코드

```cpp
#include <cstdio>
#include <algorithm>
using namespace std;

int n = 10, a[21];

int main() {
	int l, r;
	for (int i=1;i<=20;i++) a[i] = i;
	while (n--) {
		scanf("%d %d", &l, &r);
		reverse(a + l, a + r + 1);
	}
	for (int i=1;i<=20;i++) {
		printf("%d ", a[i]);
	}

	return 0;
}
```
