[14585 : 사수빈탕](https://www.acmicpc.net/problem/14585)  

## 풀이

전형적인 다이나믹  

`dp[i][j] = max(dp[i-1][j], dp[i][j-1]) + val`  

## 코드

```cpp
#include <cstdio>
#include <algorithm>
using namespace std;

int n, m, a[303][303];
long long d[303][303];

int main() {
    scanf("%d %d", &n, &m);
    for (int i = 0; i < n; i++) {
        int x, y;
        scanf("%d %d", &x, &y);
        a[x+1][y+1] = 1;
    }

    for (int i = 1; i <= 301; i++) {
        for (int j = 1; j <= 301; j++) {
            d[i][j] = max(d[i-1][j], d[i][j-1]);
            if (a[i][j] && m-i-j+2 > 0) d[i][j] += m-i-j+2;
        }
    }

    printf("%lld", d[301][301]);

    return 0;
}
```
