## 코드

```cpp
#include <cstdio>

int main() {
	int h, m, s, tc, t, c, mod = 86400;
	scanf("%d %d %d", &h, &m, &s);
	s += m * 60 + h * 3600;
	for (scanf("%d", &tc); tc; tc--) {
		scanf("%d", &t);
		if (t == 1) scanf("%d", &c), s = (s + c) % mod;
		if (t == 2) scanf("%d", &c), s = (s - c) % mod;
		if (t == 3) {
			if (s < 0) s += mod;
			printf("%d %d %d\n", s / 3600, s%3600/60, s%60);
		}
	}

	return 0;
}
```
