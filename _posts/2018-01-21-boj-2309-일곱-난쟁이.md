[2309 : 일곱 난쟁이](https://www.acmicpc.net/problem/2309)

## 풀이

돌려돌려~

## 코드

```cpp
#include <stdio.h>
#include <algorithm>

int a[10];

int main() {
	for (int i = 0; i < 9; i++) scanf("%d", &a[i]);

	std::sort(a, a + 9);

	for (int i = 0; i < (1 << 9); i++) {
		int cnt = 0, sum = 0;
		
		for (int j = 0; j < 9; j++) if ((1 << j) & i)
			cnt++, sum += a[j];

		if (cnt != 7 || sum != 100) continue;

		for (int j = 0; j < 9; j++) if ((1 << j) & i)
			printf("%d\n", a[j]);

		break;
	}

	return 0;
}
```
