## 풀이

그리워하면 언젠가 만나게 되는

## 코드

```cpp
#include <stdio.h>
#include <string.h>
int main() {
    int T, N, M, p, i, j, n[21], d[10001];
    scanf("%d", &T);
    while (T--) {
        scanf("%d", &N);
        memset(n, 0, sizeof(n)); memset(d, 0, sizeof(d));
        for (i = 1; i <= N; i++) scanf("%d", &n[i]);
        scanf("%d", &M);
        d[0] = 1;
        for (i = 1; i <= N; i++) for (j = 0; j <= M - n[i]; j++)
            d[j + n[i]] += d[j];
        printf("%d\n", d[M]);
    }
    return 0;
}
```
