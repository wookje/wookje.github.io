[15559 : 내 선물을 받아줘](https://www.acmicpc.net/problem/15559)

## 풀이

boj에 내가 나왔으면 정말 좋겠네~

정말 좋겠네~~~~

## 코드

```cpp
#include <iostream>
using namespace std;

int n, m, i, j, ans, cnt = 1, vst[1001][1001];
char a[1001][1001];

int main() {
    cin.tie(0); ios_base::sync_with_stdio(0);

    cin >> n >> m;
    for (i = 0; i < n; i++) cin >> a[i];

    for (i = 0; i < n; i++) for (j = 0; j < m; j++) {
        if (vst[i][j]) continue;
        int y = i, x = j;
        while (!vst[y][x]) {
            vst[y][x] = cnt;
            char c = a[y][x];
            if (c == 'S') y++; if (c == 'E') x++;
            if (c == 'N') y--; if (c == 'W') x--;
        }
        ans += vst[y][x] == cnt++;
    }

    cout << ans;

    return 0;
}
```
