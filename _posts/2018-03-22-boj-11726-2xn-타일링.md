## 코드

```cpp
#include <stdio.h>

int n, dp[1010];

int main() {
	dp[0] = dp[1] = 1;

	scanf("%d", &n);

	for (int i = 2; i <= n; i++)
		dp[i] = (dp[i - 1] + dp[i - 2]) % 10007;

	printf("%d", dp[n] % 10007);

	return 0;
}
```
```cpp
main(n,a,b) {
    scanf("%d",&n);
    a=b=1;
    while(n--) {
        b+=a;
        a=b-a;
        b%=10007;
    }
    printf("%d",a);
}
```
