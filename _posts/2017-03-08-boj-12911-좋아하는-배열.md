[12911 : 좋아하는 배열](https://www.acmicpc.net/problem/12911)

## 풀이

dp[i][j] = 길이가 i이고, 첫번째 숫자가 j  

나머지 연산에 주의하자.  

## 코드

```cpp
#include <stdio.h>

typedef long long ll;

const int mod = 1000000007;

int N, K;
int dp[11][100001];
ll sum[11];

int main() {
	scanf("%d %d", &N, &K);

	for (int i = 0; i <= K; ++i) dp[1][i] = 1;
	sum[1] = K;
	for (int i = 2; i <= N; ++i) {
		sum[i - 1] += mod;
		for (int j = 1; j <= K; ++j) {
			ll temp = 0;
			for (int k = j + j; k <= K; k += j) {
				temp += dp[i - 1][k];
				temp %= mod;
			}
			dp[i][j] = sum[i - 1] - temp;
			dp[i][j] %= mod;
			sum[i] += dp[i][j];
			sum[i] %= mod;
		}
	}
	printf("%lld", sum[N]);

	return 0;
}
```
