[15366 : Olivander](https://www.acmicpc.net/problem/15366)  

## 풀이

정.렬.조.아

## 코드

```cpp
#include <cstdio>
#include <algorithm>
using namespace std;
int n, a[100], b[100];
int main() {
    scanf("%d", &n);
	for (int i = 0; i < n; i++) scanf("%d", &a[i]);
	for (int i = 0; i < n; i++) scanf("%d", &b[i]);
	sort(a, a + n); sort(b, b + n);
	while (n--) if (a[n] > b[n]) return !printf("NE");
	return !printf("DA");
}
```
