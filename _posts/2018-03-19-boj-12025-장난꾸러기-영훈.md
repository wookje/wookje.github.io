[12025 : 장난꾸러기 영훈](https://www.acmicpc.net/problem/12025)

## 풀이

아 9시 수업 에반데

## 코드

```cpp
#include <stdio.h>
long long i, k;
char s[66];
int main() {
	scanf("%s %lld", s, &k); k--;
	for (i = 0; s[i]; i++) {
		if (s[i] == '6' || s[i] == '7') s[i] -= 5;
	}
	while (i--) {
		if (s[i] == '1' || s[i] == '2') {
			if (k & 1) s[i] += 5;
			k >>= 1;
		}
	}
	puts(k ? "-1" : s);
	return 0;
}
```
