[1068 : 트리](https://www.acmicpc.net/problem/1068)  

## 풀이

잘 짜주자.

## 코드

```cpp
#include <stdio.h>

int n, i, now, del, cnt, gph[51], idg[51];

int main() {
	scanf("%d", &n);
	for (i = 0; i < n; i++) {
		scanf("%d", &gph[i]);
		if (~gph[i]) idg[gph[i]]++;
	}
	scanf("%d", &del);

	if (gph[del] == -1) return !printf("0");
	idg[gph[del]]--;
	for (i = 0; i < n; i++) {
		if (idg[i]) continue;
		cnt++;
		for (now = i; ~now; now = gph[now]) if (now == del) {
			cnt--; break;
		}
	}

	printf("%d", cnt);

	return 0;
}
```
