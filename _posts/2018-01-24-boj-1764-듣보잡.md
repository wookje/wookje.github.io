[1764 : 듣보잡](https://www.acmicpc.net/problem/1764)

## 풀이

stl로 떡칠해보자!!

## 코드

```cpp
#include <iostream>
#include <algorithm>
#include <string>
#include <vector>
#include <set>
using namespace std;

int n, m;
string t;
set<string> s;
vector<string> v;

int main() {
	cin.tie(0); ios_base::sync_with_stdio(0);

	cin >> n >> m;
	while (n--) cin >> t, s.insert(t);
	while (m--) cin >> t, s.count(t) ? v.emplace_back(t) : [](){}();

	sort(v.begin(), v.end());
	cout << v.size() << "\n";
	for (auto it : v) cout << it << "\n";

	return 0;
}
```
