[1991 : 트리 순회](https://www.acmicpc.net/problem/1991)

## 풀이

아 퇴사하고싶다

## 코드

```cpp
#include <stdio.h>
#define p(c) putchar(c);

int n, i;
char a, b, c, t[99][2];

void f(int x) {
	if (i == 0) p(x);
	if (t[x][0] != '.') f(t[x][0]);
	if (i == 1) p(x);
	if (t[x][1] != '.') f(t[x][1]);
	if (i == 2) p(x);
}

int main() {
	scanf("%d", &n);
	for (i = 0; i < n; i++) {
		scanf(" %c %c %c", &a, &b, &c);
		t[a][0] = b, t[a][1] = c;
	}
	for (i = 0; i < 3; i++) f('A'), puts("");
	return 0;
}
```
