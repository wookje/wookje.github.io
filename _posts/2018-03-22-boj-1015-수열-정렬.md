[1015 : 수열 정렬](https://www.acmicpc.net/problem/1015)

## 풀이

졸리다

## 코드

```cpp
#include <stdio.h>
#include <algorithm>
using namespace std;
int n, i, b[55];
pair<int, int> a[55];
int main() {
	scanf("%d", &n);
	while (i < n) scanf("%d", &a[i].first), a[i].second = i, i++;
    sort(a, a + n);
	while (i--) b[a[i].second] = i;
	i++;
	while (i < n) printf("%d ", b[i++]);
	return 0;
}
```
