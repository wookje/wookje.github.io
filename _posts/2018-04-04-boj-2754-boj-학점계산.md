## 코드

```cpp
#include <cstdio>
int tc;
double r;
char s[3];
int main() {
    scanf("%s", s);
    if (s[0] == 'A') r = 4; if (s[0] == 'C') r = 2; 
    if (s[0] == 'B') r = 3; if (s[0] == 'D') r = 1;
    if (s[1] == '+') r += 0.3; if (s[1] == '-') r -= 0.3;
    printf("%.1lf", r);
    return 0;
}
```
