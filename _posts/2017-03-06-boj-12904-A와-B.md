[12904번: A와 B](https://www.acmicpc.net/problem/12904)

## 풀이

직접 시뮬레이션 해보면 풀 수 있다!  

## 코드

```cpp
#include <bits/stdc++.h>
using namespace std;

string S, T;

void reverse(string K) {
	for (int i = 0; i < K.length(); ++i)
		T[K.length() - i - 1] = K[i];
}

int main() {
	cin >> S >> T;
	
	int flag = 0;
	while (true) {
		if (S.length() == T.length()) {
			if (!S.compare(T)) flag = 1;
			break;
		}
		int len = T.length() - 1;
		char now = T[len];
		T.erase(len);
		if (now == 'B')
			reverse(T);
	}
	printf("%d", flag);

	return 0;
}
```
