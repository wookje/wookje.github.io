[2623 : 음악프로그램](https://www.acmicpc.net/problem/2623)

## 풀이

위상정렬을 하면 된다!  

## 코드

```cpp
#include <iostream>
#include <vector>
using namespace std;

const int n_ = 1000 + 3;

int n, m, idg[n_];
vector<int> ans;
vector<int> edg[n_];

void dfs(int now) {
	idg[now] = -1;
	ans.push_back(now);
	for (auto nxt : edg[now])
		if (!(--idg[nxt])) dfs(nxt);
}

int main() {
	int i, j, a, b, c;
	cin >> n >> m;
	while (m--) {
		cin >> a >> b;
		while (--a) {
			cin >> c;
			edg[b].push_back(c);
			idg[c]++;
			b = c;
		}
	}
	for (i = 1; i <= n; i++)
		if (!idg[i]) dfs(i);
	
	if (ans.size() < n) puts("0");
	else for (auto p : ans) printf("%d\n", p);

	return 0;
}
```
