## 코드

```cpp
#include <cstdio>
#include <algorithm>
using namespace std;

int n, t, h, a[3003], dp[3003];

int main() {
    scanf("%d", &n);
    for (int i = 1; i <= n; i++) scanf("%d", &a[i]);
    scanf("%d %d", &t, &h);

    sort(a + 1, a + 1 + n);

    for (int i = 1; i <= n; i++) {
        dp[i] = dp[i - 1] + a[i] * t;
        int cst = h;
        for (int j = i; j; j--) {
            cst += (a[(i + j + 1) / 2] - a[j]) * t;
            dp[i] = min(dp[i], dp[j - 1] + cst);
        }
    }

    printf("%d ", dp[n]);

    return 0;
}
```
