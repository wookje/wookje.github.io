## 코드

```cpp
#include <stdio.h>
#include <algorithm>

int n, x, y, a[4] = { 0,1 };

int main() {
	scanf("%d", &n);
	while (n--) scanf("%d %d", &x, &y), std::swap(a[x], a[y]);
	for (n = 1; n <= 3; n++) if (a[n]) printf("%d", n);
	return 0;
}
```
