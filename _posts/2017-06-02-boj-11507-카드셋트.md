## 코드

```cpp
#include <stdio.h>
int a, b, cnt[4];
bool chk[14][4];
int main() {
	for (char c; ~scanf("%c%1d%1d", &c, &a, &b);) {
		a = a * 10 + b;
		switch (c) {
		case 'P': c = 0; break;
		case 'K': c = 1; break;
		case 'H': c = 2; break;
		case 'T': c = 3;
		}
		if (chk[a][c]) { puts("GRESKA"); return 0; }
		chk[a][c] = 1, cnt[c]++;
	}
	for (int i = 0; i < 4; i++) printf("%d ", 13 - cnt[i]);

	return 0;
}
```
