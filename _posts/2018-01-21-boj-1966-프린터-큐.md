[1966 : 프린터 큐](https://www.acmicpc.net/problem/1966)

## 풀이

프린터 터져욧!!

## 코드

```cpp
#include </bits/stdc++.h>
using namespace std;

int main() {
	int tc;
	scanf("%d", &tc);
	while (tc--) {
		int n, m, cnt = 1;
		queue<pair<int, int> > qu;
		priority_queue<int> pq;

		scanf("%d %d", &n, &m);
		for (int i = 0, a; i < n; i++) {
			scanf("%d", &a);
			qu.push({ a,i });
			pq.push(a);
		}

		while (!pq.empty()) {
			auto n1 = qu.front();
			int n2 = pq.top();
			if (n1.first == n2) {
				if (n1.second == m) {
					printf("%d\n", cnt);
					break;
				}
				pq.pop(), cnt++;
			}
			else qu.push(n1);
			qu.pop();
		}
	}

	return 0;
}
```
